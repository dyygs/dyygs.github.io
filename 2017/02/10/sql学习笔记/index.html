SQL: Structured Query Language

SELECT

SELECT COLUME_NAME FROM TABLE_NAME

DISTINCT

SELECT DISTINCT COLUME_NAME FROM TABLE_NAME

WHERE

SELECT COLUME_NAME FROM TABLE_NAME WHERE COLUME_NAME OPERATOR VALUE
= , <> , > , < , >= , <= , BETWEEN , LIKE

AND & OR

SELECT COLUME_NAME FROM TABLE_NAME WHERE COLUME_NAME OPERATOR VALUE AND(OR) COLUME_NAME OPERATOR VALUE

ORDER BY

SELECT COLUME_NAME FROM TABLE_NAME ORDER BY COLUME_NAME DESC, COLUME_NAME ASC

INSERT INTO

INSERT INTO TABLE_NAME VALUES(VALUE1, VALUE2, …)
INSERT INTO TABLE_NAME (COLUME_NAME1, COLUME_NAME2) VALUES (VALUE1, VALUE2)

UPDATE

UPDATE TABLE_NAME SET COLUME_NAME = VALUE WHERE COLUME OPERATOR VALUE

DELETE

DELETE FROM TABLE_NAME WHERE COLUME_NAME OPERATOR VALUE

TOP

SELECT TOP NUMBER|PERCENT COLUME_NAME FROME TABLE_NAME

LIKE

SELECT COLUME_NAME FROM TABLE_NAME WHERE COLUME_NAME LIKE ‘%PATTERN%’

% 一个或多个字符 _ 一个字符 [charlist] 字符中的任何一个单字符 [!charlist] 不在字符列中的单字符

从 “Persons” 表中选取居住的城市不以 “A” 或 “L” 或 “N” 开头的人：
我们可以使用下面的 SELECT 语句：

SELECT * FROM Persons
WHERE City LIKE ‘[!ALN]%’

IN

SELECT COLUME_NAME FROM TABLE_NAME WHERE COLUME_NAME IN (VALUE1, VALUE2,..)

选取姓氏为 Adams 和 Carter 的人：
我们可以使用下面的 SELECT 语句：

SELECT * FROM Persons
WHERE LastName IN (‘Adams’,’Carter’)

BETWEEN

SELECT COLUME_NAME FROM TABLE_NAME WHERE COLUME_NAME (NOT) BETWEEN VALUE1 AND VALUE2

AS

SELECT * FROM TABLE_NAME AS ALIAS_NAME WHERE ALIAS_NAME.COLUME_NAME OPERATOR VALUE

列名的别名

SELECT COLUME_NAME AS ALIAS_NAME FROM TABLE_NAME

INNER JOIN

SELECT column_name(s)
FROM table_name1
INNER JOIN table_name2
ON table_name1.column_name=table_name2.column_name

等价于

SELECT COLUME_NAME(S) FROM TABLE_NAME1, TABLE_NAME2 WHERE TABLE_NAME1.COLUME_NAME = TABLE_NAME2.COLUME_NAME

LEFT JOIN (RIGHT JOIN, FULL JOIN)

LEFT JOIN 关键字会从左表 (table_name1) 那里返回所有的行，即使在右表 (table_name2) 中没有匹配的行

SELECT column_name(s)
FROM table_name1
LEFT JOIN table_name2
ON table_name1.column_name=table_name2.column_name

CREATE

CREATE TABLE TABLE_NAME {
COLUME1, DATA_TYPE,
COLUME2, DATA_TYPE
}

integer(size) int(size) smallint(size) tinyint(size)
仅容纳整数。在括号内规定数字的最大位数。

decimal(size,d) numeric(size,d)
容纳带有小数的数字。
“size” 规定数字的最大位数。”d” 规定小数点右侧的最大位数。

char(size)
容纳固定长度的字符串（可容纳字母、数字以及特殊字符）。
在括号中规定字符串的长度。

varchar(size)
容纳可变长度的字符串（可容纳字母、数字以及特殊的字符）。
在括号中规定字符串的最大长度。

date(yyyymmdd) 容纳日期。

CONSTRAINTS 约束

  • NOT NULL 不接受空值
  • UNIQUE 唯一
  • PRIMARY KEY
  • FOREIGN KEY
  • CHECK 检查条件
  • DEFAULT 默认值
CREATE INDEX

CREATE INDEX index_name
ON table_name (column_name)
